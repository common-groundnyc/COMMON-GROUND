Strategy: Stage a temp table of unique corp-like names from ALL sources first, then do ONE match against nys_corporations. This is far more efficient than multiple IN subqueries.

```python
# Stage 1: Collect all corp-like names from every source
conn.execute("""
    CREATE OR REPLACE TEMP TABLE _corp_name_targets AS
    -- Source 1: HPD/PLUTO owners (existing)
    SELECT DISTINCT UPPER(TRIM(owner_name)) AS target_name
    FROM main.graph_owners
    WHERE owner_name IS NOT NULL AND LENGTH(owner_name) > 3
      AND (UPPER(owner_name) LIKE '%LLC%' OR UPPER(owner_name) LIKE '%CORP%'
           OR UPPER(owner_name) LIKE '%INC%' OR UPPER(owner_name) LIKE '%LP%'
           OR UPPER(owner_name) LIKE '%LTD%' OR UPPER(owner_name) LIKE '%CO.%'
           OR UPPER(owner_name) LIKE '%REALTY%' OR UPPER(owner_name) LIKE '%MGMT%'
           OR UPPER(owner_name) LIKE '%MANAGEMENT%' OR UPPER(owner_name) LIKE '%PROPERTIES%'
           OR UPPER(owner_name) LIKE '%ASSOCIATES%' OR UPPER(owner_name) LIKE '%PARTNERS%')
    UNION
    -- Source 2: ACRIS party names (grantees/grantors with corp indicators)
    SELECT DISTINCT UPPER(TRIM(name)) AS target_name
    FROM lake.housing.acris_parties
    WHERE name IS NOT NULL AND LENGTH(TRIM(name)) > 3
      AND (UPPER(name) LIKE '%LLC%' OR UPPER(name) LIKE '%CORP%'
           OR UPPER(name) LIKE '%INC%' OR UPPER(name) LIKE '%LP%'
           OR UPPER(name) LIKE '%LTD%')
    UNION
    -- Source 3: OATH respondent names (corps facing enforcement)
    SELECT DISTINCT UPPER(TRIM(respondent_name)) AS target_name
    FROM lake.city_government.oath_hearings
    WHERE respondent_name IS NOT NULL AND LENGTH(TRIM(respondent_name)) > 3
      AND (UPPER(respondent_name) LIKE '%LLC%' OR UPPER(respondent_name) LIKE '%CORP%'
           OR UPPER(respondent_name) LIKE '%INC%')
    UNION
    -- Source 4: Campaign employer names
    SELECT DISTINCT UPPER(TRIM(empname)) AS target_name
    FROM lake.financial.campaign_contributions
    WHERE empname IS NOT NULL AND LENGTH(TRIM(empname)) > 3
      AND (UPPER(empname) LIKE '%LLC%' OR UPPER(empname) LIKE '%CORP%'
           OR UPPER(empname) LIKE '%INC%' OR UPPER(empname) LIKE '%LP%')
""")
```

Then the graph_corps query becomes:
```sql
CREATE OR REPLACE TABLE main.graph_corps AS
WITH relevant_corps AS (
    SELECT c.dos_id, c.current_entity_name, c.entity_type,
           c.initial_dos_filing_date, c.county,
           c.dos_process_name, c.dos_process_address_1, c.dos_process_city,
           c.registered_agent_name, c.chairman_name
    FROM lake.business.nys_corporations c
    WHERE c.current_entity_name IS NOT NULL
      AND UPPER(TRIM(c.current_entity_name)) IN (SELECT target_name FROM _corp_name_targets)
)
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (PARTITION BY dos_id ORDER BY initial_dos_filing_date DESC NULLS LAST) AS rn
    FROM relevant_corps
) WHERE rn = 1
```
