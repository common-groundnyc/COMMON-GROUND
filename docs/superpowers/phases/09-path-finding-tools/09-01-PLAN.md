**DuckPGQ MATCH gotchas (CRITICAL):**
- MATCH inside CTEs/UNION **segfaults** — NEVER do this
- Only ANY SHORTEST works (no ALL SHORTEST, no Top-K)
- Left-directed paths don't work — only right-directed or undirected
- Edge variables REQUIRED: `[e:Label]` not `[:Label]`
- PageRank uses labels not table names: `pagerank(graph, vertex_label, edge_label)`

**DuckPGQ MATCH syntax (tested working in Phase 8):**
```sql
SELECT * FROM GRAPH_TABLE(nyc_unified
    MATCH (o:Owner)-[e:OwnerIsTransactor]->(t:TxEntity)
    COLUMNS (o.owner_name, t.entity_name, t.tx_count)
) LIMIT 10
```

**ANY SHORTEST syntax (from DuckPGQ docs):**
```sql
SELECT * FROM GRAPH_TABLE(graph_name
    MATCH ANY SHORTEST (a:Label WHERE a.key = 'value')-[e:*1..5]->(b:Label WHERE b.key = 'value2')
    COLUMNS (a.key AS source, b.key AS target, path_length(e) AS hops)
)
```

**Available graphs for path-finding:**
- nyc_unified: 5 vertex types, 5 edge types (best for cross-domain paths)
- nyc_building_network: Building → SharedOwner → Building
- nyc_corporate_web: Corp + Person → OfficerOf/SharedOfficer

**SQL fallback strategy**: If MATCH segfaults on a particular query, fall back to iterative BFS using regular SQL JOINs on the bridge tables.

### Tool 1: shortest_path
```python
@mcp.tool()
async def shortest_path(source_name: str, target_name: str, max_hops: int = 5) -> str:
    """Find the shortest connection path between two entities across all NYC domains (housing, corporate, transactions, politics). Uses the unified cross-domain graph. Example: shortest_path('KUSHNER COMPANIES LLC', 'DONALD J TRUMP')"""
    try:
        conn = _get_conn()
        source = source_name.upper().strip()
        target = target_name.upper().strip()

        # Try unified graph MATCH first
        try:
            # Check which vertex tables contain the entities
            source_in = {}
            target_in = {}
            for label, table, key in [
                ('Owner', 'graph_owners', 'owner_id'),
                ('TxEntity', 'graph_tx_entities', 'entity_name'),
                ('Corp', 'graph_corps', 'current_entity_name'),
                ('PoliticalEntity', 'graph_pol_entities', 'entity_name'),
            ]:
                s = conn.execute(f"SELECT COUNT(*) FROM main.{table} WHERE UPPER(TRIM({key})) = ?", [source]).fetchone()[0]
                t = conn.execute(f"SELECT COUNT(*) FROM main.{table} WHERE UPPER(TRIM({key})) = ?", [target]).fetchone()[0]
                if s > 0: source_in[label] = (table, key)
                if t > 0: target_in[label] = (table, key)

            if not source_in:
                return f"'{source_name}' not found in any graph vertex table."
            if not target_in:
                return f"'{target_name}' not found in any graph vertex table."

            # Try direct bridge table lookup (SQL fallback — faster and safer than MATCH)
            paths = []
            # Check direct connection via bridge tables
            for bridge, src_col, dst_col, src_label, dst_label in [
                ('graph_owner_tx_bridge', 'owner_id', 'tx_entity_name', 'Owner', 'TxEntity'),
                ('graph_owner_pol_bridge', 'owner_id', 'pol_entity_name', 'Owner', 'PoliticalEntity'),
                ('graph_corp_pol_bridge', 'dos_id', 'pol_entity_name', 'Corp', 'PoliticalEntity'),
                ('graph_corp_tx_bridge', 'dos_id', 'tx_entity_name', 'Corp', 'TxEntity'),
            ]:
                # Source → Bridge → Target (1 hop)
                r = conn.execute(f"""
                    SELECT {src_col}, {dst_col} FROM main.{bridge}
                    WHERE {src_col} = ? AND {dst_col} = ?
                """, [source, target]).fetchone()
                if r:
                    paths.append(f"{source} --[{bridge}]--> {target} (1 hop)")

                # Source → Bridge → Shared → Bridge → Target (2 hops)
                r = conn.execute(f"""
                    SELECT a.{src_col}, a.{dst_col}, b.{src_col}
                    FROM main.{bridge} a
                    JOIN main.{bridge} b ON a.{dst_col} = b.{dst_col}
                    WHERE a.{src_col} = ? AND b.{src_col} = ?
                    LIMIT 1
                """, [source, target]).fetchone()
                if r:
                    paths.append(f"{source} --[{bridge}]--> {r[1]} --[{bridge}]--> {target} (2 hops)")

            if paths:
                return f"Paths found between '{source_name}' and '{target_name}':\n" + "\n".join(paths)

            # List what domains each entity is in
            src_domains = ", ".join(source_in.keys())
            tgt_domains = ", ".join(target_in.keys())
            return f"No direct path found (max {max_hops} hops).\n{source_name}: [{src_domains}]\n{target_name}: [{tgt_domains}]"

        except Exception as match_err:
            return f"Path search failed: {match_err}"
    except Exception as e:
        return f"Error: {e}"
```

### Tool 2: entity_reachable
```python
@mcp.tool()
async def entity_reachable(name: str) -> str:
    """Show all domains and connections for an entity. Lists every graph where the entity appears and counts of cross-domain bridges."""
    try:
        conn = _get_conn()
        entity = name.upper().strip()

        results = []
        # Check each vertex table
        checks = [
            ('Housing (Owner)', 'graph_owners', 'owner_id', 'owner_name'),
            ('Transaction', 'graph_tx_entities', 'entity_name', 'entity_name'),
            ('Corporate', 'graph_corps', 'dos_id', 'current_entity_name'),
            ('Political', 'graph_pol_entities', 'entity_name', 'entity_name'),
        ]
        found_in = []
        for domain, table, key, name_col in checks:
            r = conn.execute(f"SELECT COUNT(*) FROM main.{table} WHERE UPPER(TRIM({name_col})) = ?", [entity]).fetchone()[0]
            if r > 0:
                found_in.append(domain)
                results.append(f"  {domain}: YES")
            else:
                results.append(f"  {domain}: no")

        # Count bridge connections
        bridges = []
        bridge_checks = [
            ('Owner→Transaction', 'graph_owner_tx_bridge', 'owner_id'),
            ('Owner→Political', 'graph_owner_pol_bridge', 'owner_id'),
            ('Corp→Political', 'graph_corp_pol_bridge', 'dos_id'),
            ('Corp→Transaction', 'graph_corp_tx_bridge', 'dos_id'),
        ]
        for label, table, col in bridge_checks:
            try:
                r = conn.execute(f"SELECT COUNT(*) FROM main.{table} WHERE {col} = ?", [entity]).fetchone()[0]
                if r > 0:
                    bridges.append(f"  {label}: {r} connections")
            except:
                pass

        header = f"Entity reachability for '{name}':"
        domain_section = f"\nDomains ({len(found_in)}/4):\n" + "\n".join(results)
        bridge_section = f"\nCross-domain bridges:\n" + ("\n".join(bridges) if bridges else "  None")

        return header + domain_section + bridge_section
    except Exception as e:
        return f"Error: {e}"
```

**Key design decisions:**
- Use SQL-based path finding (bridge table lookups) instead of DuckPGQ MATCH for shortest_path — more reliable, avoids segfault risk
- DuckPGQ MATCH can be added as an enhancement in Phase 10 once we know which patterns are safe
- entity_reachable is a diagnostic tool showing cross-domain presence
