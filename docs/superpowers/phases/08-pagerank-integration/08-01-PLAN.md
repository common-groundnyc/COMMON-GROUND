### Tool 1: landlord_pagerank
```python
@mcp.tool()
async def landlord_pagerank(borough: str = None, top_n: int = 25) -> str:
    """Rank buildings by PageRank in the shared-owner network. Buildings connected to many other buildings through shared landlords score highest. Optional borough filter (1=Manhattan, 2=Bronx, 3=Brooklyn, 4=Queens, 5=Staten Island)."""
    try:
        conn = _get_conn()
        where = f"WHERE b.boroid = '{borough}'" if borough else ""
        rows = conn.execute(f"""
            WITH pr AS (
                SELECT * FROM pagerank(nyc_building_network, graph_buildings, bbl)
            )
            SELECT pr.bbl, pr.pagerank,
                   b.housenumber, b.streetname, b.zip, b.boroid,
                   b.total_units, b.stories
            FROM pr
            JOIN main.graph_buildings b ON pr.bbl = b.bbl
            {where}
            ORDER BY pr.pagerank DESC
            LIMIT {top_n}
        """).fetchall()
        # Format results
        lines = [f"Top {len(rows)} buildings by PageRank (shared-owner network):"]
        for r in rows:
            lines.append(f"  BBL {r[0]} (rank: {r[1]:.6f}) — {r[2]} {r[3]}, {r[4]} | {r[6]} units, {r[7]} stories")
        return "\n".join(lines)
    except Exception as e:
        return f"PageRank error: {e}"
```

### Tool 2: entity_influence
```python
@mcp.tool()
async def entity_influence(domain: str = "unified", top_n: int = 25) -> str:
    """Rank entities by PageRank across domain graphs. Domain options: 'building' (shared-owner network), 'corporate' (shared-officer network), 'transaction' (co-transaction network), 'political' (donation/lobbying network), 'unified' (cross-domain). Default: unified."""
    try:
        conn = _get_conn()
        graph_map = {
            "building": ("nyc_building_network", "graph_buildings", "bbl"),
            "corporate": ("nyc_corporate_web", "graph_corps", "dos_id"),
            "transaction": ("nyc_transaction_network", "graph_tx_entities", "entity_name"),
            "political": ("nyc_influence_network", "graph_pol_entities", "entity_name"),
            "unified": ("nyc_unified", "graph_owners", "owner_id"),
        }
        if domain not in graph_map:
            return f"Unknown domain '{domain}'. Options: {', '.join(graph_map.keys())}"
        graph, vtable, vkey = graph_map[domain]
        rows = conn.execute(f"""
            SELECT * FROM pagerank({graph}, {vtable}, {vkey})
            ORDER BY pagerank DESC
            LIMIT {top_n}
        """).fetchall()
        lines = [f"Top {len(rows)} entities by PageRank ({domain} network):"]
        for r in rows:
            lines.append(f"  {r[0]} — PageRank: {r[1]:.6f}")
        return "\n".join(lines)
    except Exception as e:
        return f"PageRank error: {e}"
```

**Implementation notes:**
- The PageRank function is a DuckPGQ scalar function that returns a table
- It may take 10-30 seconds on large graphs — that's normal
- If unified graph PageRank OOMs, fall back to individual domain graphs only
- Wrap in try/except since PageRank can fail on disconnected graphs
