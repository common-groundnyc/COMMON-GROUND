```python
# Stage 1: Deed parties (existing logic)
conn.execute("""
    CREATE OR REPLACE TEMP TABLE _tx_deed_parties AS
    SELECT UPPER(TRIM(p.name)) AS entity_name, p.party_type,
           (l.borough || LPAD(l.block::VARCHAR, 5, '0') || LPAD(l.lot::VARCHAR, 4, '0')) AS bbl,
           TRY_CAST(m.document_amt AS DOUBLE) AS amount,
           TRY_CAST(m.document_date AS DATE) AS doc_date,
           m.document_id, m.doc_type
    FROM lake.housing.acris_parties p
    JOIN lake.housing.acris_master m ON p.document_id = m.document_id
    JOIN lake.housing.acris_legals l ON m.document_id = l.document_id
    WHERE m.doc_type IN ('DEED', 'DEEDO', 'DEED, RP')
      AND p.name IS NOT NULL AND LENGTH(TRIM(p.name)) > 3
      AND l.borough IS NOT NULL
""")

# Stage 2: Mortgage parties
conn.execute("""
    CREATE OR REPLACE TEMP TABLE _tx_mtge_parties AS
    SELECT UPPER(TRIM(p.name)) AS entity_name, p.party_type,
           (l.borough || LPAD(l.block::VARCHAR, 5, '0') || LPAD(l.lot::VARCHAR, 4, '0')) AS bbl,
           TRY_CAST(m.document_amt AS DOUBLE) AS amount,
           TRY_CAST(m.document_date AS DATE) AS doc_date,
           m.document_id, m.doc_type
    FROM lake.housing.acris_parties p
    JOIN lake.housing.acris_master m ON p.document_id = m.document_id
    JOIN lake.housing.acris_legals l ON m.document_id = l.document_id
    WHERE m.doc_type = 'MTGE'
      AND p.name IS NOT NULL AND LENGTH(TRIM(p.name)) > 3
      AND l.borough IS NOT NULL
""")

# Stage 3: Assignment + satisfaction parties
conn.execute("""
    CREATE OR REPLACE TEMP TABLE _tx_asst_parties AS
    SELECT UPPER(TRIM(p.name)) AS entity_name, p.party_type,
           (l.borough || LPAD(l.block::VARCHAR, 5, '0') || LPAD(l.lot::VARCHAR, 4, '0')) AS bbl,
           TRY_CAST(m.document_amt AS DOUBLE) AS amount,
           TRY_CAST(m.document_date AS DATE) AS doc_date,
           m.document_id, m.doc_type
    FROM lake.housing.acris_parties p
    JOIN lake.housing.acris_master m ON p.document_id = m.document_id
    JOIN lake.housing.acris_legals l ON m.document_id = l.document_id
    WHERE m.doc_type IN ('ASST', 'SAT', 'AGMT')
      AND p.name IS NOT NULL AND LENGTH(TRIM(p.name)) > 3
      AND l.borough IS NOT NULL
""")

# Combine into entities
conn.execute("""
    CREATE OR REPLACE TABLE main.graph_tx_entities AS
    WITH all_parties AS (
        SELECT * FROM _tx_deed_parties
        UNION ALL SELECT * FROM _tx_mtge_parties
        UNION ALL SELECT * FROM _tx_asst_parties
    ),
    entity_stats AS (
        SELECT entity_name,
               COUNT(DISTINCT document_id) AS tx_count,
               COUNT(DISTINCT bbl) AS property_count,
               COUNT(*) FILTER (WHERE party_type = '1') AS as_seller,
               COUNT(*) FILTER (WHERE party_type = '2') AS as_buyer,
               MIN(doc_date) AS first_tx,
               MAX(doc_date) AS last_tx,
               SUM(amount) AS total_amount
        FROM all_parties
        GROUP BY entity_name
        HAVING COUNT(DISTINCT document_id) >= 2
    )
    SELECT * FROM entity_stats
""")

# Edges: entity → BBL
conn.execute("""
    CREATE OR REPLACE TABLE main.graph_tx_edges AS
    WITH all_parties AS (
        SELECT * FROM _tx_deed_parties
        UNION ALL SELECT * FROM _tx_mtge_parties
        UNION ALL SELECT * FROM _tx_asst_parties
    )
    SELECT DISTINCT
        entity_name, bbl,
        CASE WHEN party_type = '1' THEN 'GRANTOR'
             WHEN party_type = '2' THEN 'GRANTEE' ELSE 'OTHER' END AS role,
        amount, doc_date, document_id
    FROM all_parties
    WHERE entity_name IN (SELECT entity_name FROM main.graph_tx_entities)
""")

# Shared: entity↔entity on same document (with mega-entity cap)
conn.execute("""
    CREATE OR REPLACE TABLE main.graph_tx_shared AS
    WITH eligible_entities AS (
        SELECT entity_name FROM main.graph_tx_entities
        WHERE tx_count <= 500
    ),
    doc_entities AS (
        SELECT DISTINCT document_id, entity_name, role
        FROM main.graph_tx_edges
        WHERE entity_name IN (SELECT entity_name FROM eligible_entities)
    )
    SELECT a.entity_name AS entity1, b.entity_name AS entity2,
           COUNT(DISTINCT a.document_id) AS shared_docs,
           FIRST(a.role) AS role1, FIRST(b.role) AS role2
    FROM doc_entities a
    JOIN doc_entities b ON a.document_id = b.document_id
        AND a.entity_name < b.entity_name
    GROUP BY a.entity_name, b.entity_name
""")
```
