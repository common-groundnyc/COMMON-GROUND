### Tool: graph_health
Add a diagnostic tool that reports graph table stats:
```python
@mcp.tool()
async def graph_health() -> str:
    """Report health of all graph tables: row counts, orphan checks, property graph status."""
    try:
        conn = _get_conn()
        tables = [
            'graph_buildings', 'graph_owners', 'graph_owns', 'graph_shared_owner',
            'graph_violations', 'graph_has_violation',
            'graph_corps', 'graph_corp_people', 'graph_corp_officer_edges', 'graph_corp_shared_officer',
            'graph_tx_entities', 'graph_tx_edges', 'graph_tx_shared',
            'graph_pol_entities', 'graph_pol_donations', 'graph_pol_contracts', 'graph_pol_lobbying',
            'graph_owner_tx_bridge', 'graph_owner_pol_bridge', 'graph_corp_pol_bridge', 'graph_corp_tx_bridge',
            'graph_unified_entities',
        ]
        lines = ["Graph Health Report:"]
        total = 0
        for t in tables:
            try:
                cnt = conn.execute(f"SELECT COUNT(*) FROM main.{t}").fetchone()[0]
                lines.append(f"  {t}: {cnt:,}")
                total += cnt
            except:
                lines.append(f"  {t}: MISSING")

        lines.append(f"\nTotal rows: {total:,}")

        # Orphan checks
        orphans = conn.execute("""
            SELECT COUNT(*) FROM main.graph_owns
            WHERE owner_id NOT IN (SELECT owner_id FROM main.graph_owners)
        """).fetchone()[0]
        lines.append(f"\nOrphan checks:")
        lines.append(f"  graph_owns → graph_owners: {orphans} orphans")

        orphans2 = conn.execute("""
            SELECT COUNT(*) FROM main.graph_owns
            WHERE bbl NOT IN (SELECT bbl FROM main.graph_buildings)
        """).fetchone()[0]
        lines.append(f"  graph_owns → graph_buildings: {orphans2} orphans")

        # Property graph status
        graphs = ['nyc_housing', 'nyc_building_network', 'nyc_transaction_network',
                  'nyc_corporate_web', 'nyc_influence_network', 'nyc_unified']
        lines.append(f"\nProperty graphs:")
        for g in graphs:
            try:
                conn.execute(f"SELECT 1 FROM GRAPH_TABLE({g} MATCH (n) COLUMNS (1 AS x)) LIMIT 1")
                lines.append(f"  {g}: OK")
            except Exception as e:
                lines.append(f"  {g}: FAILED ({e})")

        return "\n".join(lines)
    except Exception as e:
        return f"Health check error: {e}"
```
