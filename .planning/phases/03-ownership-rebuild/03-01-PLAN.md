```sql
        bbl,
        address AS housenumber,
        NULL AS streetname,
        zipcode AS zip,
        borocode AS boroid,

        TRY_CAST(numfloors AS INTEGER) AS stories
    FROM lake.city_government.pluto
    WHERE bbl IS NOT NULL AND LENGTH(bbl) = 10
),
combined AS (
    SELECT * FROM hpd
    UNION ALL
    SELECT * FROM pluto
)
SELECT
    bbl,
    FIRST(housenumber) AS housenumber,
    FIRST(streetname) AS streetname,
    FIRST(zip) AS zip,
    FIRST(boroid) AS boroid,
    MAX(total_units) AS total_units,
    MAX(stories) AS stories
FROM combined
GROUP BY bbl
```

Replace the graph_owners query (line ~893) to use owner_name as the primary key instead of registrationid:

```sql
CREATE OR REPLACE TABLE main.graph_owners AS
WITH hpd_owners AS (
    SELECT DISTINCT
        UPPER(TRIM(COALESCE(
            corporationname,
            NULLIF(TRIM(COALESCE(firstname, '') || ' ' || COALESCE(lastname, '')), '')
        ))) AS owner_name
    FROM lake.housing.hpd_registration_contacts
    WHERE type IN ('CorporateOwner', 'IndividualOwner')
      AND (corporationname IS NOT NULL OR lastname IS NOT NULL)
),
pluto_owners AS (
    SELECT DISTINCT UPPER(TRIM(ownername)) AS owner_name
    FROM lake.city_government.pluto
    WHERE ownername IS NOT NULL AND TRIM(ownername) != ''
),
all_owners AS (
    SELECT owner_name FROM hpd_owners
    UNION
    SELECT owner_name FROM pluto_owners
)
SELECT
    owner_name AS owner_id,
    owner_name
FROM all_owners
WHERE owner_name IS NOT NULL AND owner_name != ''
```

Key design decisions:
- `owner_id` = `owner_name` (UPPER + TRIM normalized). This is the PK for the property graph.
- UNION (not UNION ALL) deduplicates owners that appear in both HPD and PLUTO.
- registrationid column is dropped — it was the wrong key for cross-building matching.
- The `owner_id` alias is kept so downstream property graph definitions don't need to change the column name reference.

Also replace the graph_owns query (line ~941) to link by name:

```sql
CREATE OR REPLACE TABLE main.graph_owns AS
WITH hpd_owns AS (
    SELECT DISTINCT
        UPPER(TRIM(COALESCE(
            c.corporationname,
            NULLIF(TRIM(COALESCE(c.firstname, '') || ' ' || COALESCE(c.lastname, '')), '')
        ))) AS owner_id,
        (j.boroid || LPAD(j.block::VARCHAR, 5, '0') || LPAD(j.lot::VARCHAR, 4, '0')) AS bbl
    FROM lake.housing.hpd_jurisdiction j
    JOIN lake.housing.hpd_registration_contacts c
        ON j.registrationid::VARCHAR = c.registrationid::VARCHAR
    WHERE c.type IN ('CorporateOwner', 'IndividualOwner')
      AND j.boroid IS NOT NULL
      AND (c.corporationname IS NOT NULL OR c.lastname IS NOT NULL)
),
pluto_owns AS (
    SELECT DISTINCT
        UPPER(TRIM(ownername)) AS owner_id,
        bbl
    FROM lake.city_government.pluto
    WHERE ownername IS NOT NULL AND TRIM(ownername) != ''
      AND bbl IS NOT NULL AND LENGTH(bbl) = 10
)
SELECT owner_id, bbl
FROM (
    SELECT * FROM hpd_owns
    UNION
    SELECT * FROM pluto_owns
)
WHERE owner_id IS NOT NULL AND owner_id != ''
  AND bbl IN (SELECT bbl FROM main.graph_buildings)
```

The bbl filter against graph_buildings prevents orphaned edges (established pattern from 02-02).
